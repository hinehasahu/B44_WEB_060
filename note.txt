  email,

  location,

  updatedAt

CrimeReport Schema:-
{
  _id,
  reportedBy: ObjectId (User),
  incidentType: String,
  date: Date,
  time: String,
  location: String,
  description: String,
  evidenceFiles: [String], // file URLs
  witnesses: [ObjectId], // references
  lawyerComments: [{ lawyerId, comment, timestamp }],
  hearingRecords: [{
  date: { type: Date, required: true },
  notes: { type: String },
  audioFile: { type: String },
  isFinalHearing: { type: Boolean, default: false }
  }],
  verdict: {
    judgeName: { type: String },
    decision: { type: String },
    reasoning: { type: String },
    decisionDate: { type: Date }
  },
  status: "Pending" | "Under Investigation" | "Closed",
  caseComplexity: String,
  aiPrediction: String,
  verdict: {
    judgeName,
    decision,
    reasoning,
    decisionDate
  },
  shareLinks: {
    public: String,
    private: String
  },
  createdAt,
  updatedAt
}
